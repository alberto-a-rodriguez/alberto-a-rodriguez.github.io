Feed Monitoring & Status Rollup

Deterministic Record Selection Pick one "best" record per entity, reproducibly, using a prioritized and fully deterministic tie-break hierarchy. 
-- Goal: pick ONE "best" record per customer per year, deterministically.
-- Pattern: QUALIFY ROW_NUMBER() with a prioritized, fully deterministic ORDER BY.

SELECT
    src.customer_id,
    src.entry_year,
    src.address_type,
    src.region_code
FROM staging.customer_addresses AS src
QUALIFY ROW_NUMBER() OVER (
    PARTITION BY src.customer_id, src.entry_year
    ORDER BY
        DECODE(src.address_type, 'permanent', 1, 'mailing', 2, 3) ASC,  -- categorical priority
        CASE WHEN src.region_code IS NOT NULL THEN 0 ELSE 1 END ASC,    -- prefer complete data
        src.updated_at DESC NULLS LAST,                                 -- most recent wins
        src.record_hash ASC                                             -- deterministic final tiebreak
) = 1;

Guarantees one stable row per entity even when inputs tie.

Multi-Source Standardization Unify many sources into one consistent attribute schema so new sources plug in without reshaping downstream models. 
-- Goal: unify many sources into ONE consistent attribute schema (EAV style).
-- Pattern: modular CTEs + UNION ALL with a shared output contract.

WITH academic_attributes AS (
    SELECT customer_id, entry_year, 'academic' AS attribute_group,
           attribute_type, attribute_value
    FROM staging.source_academic
),
financial_attributes AS (
    SELECT customer_id, entry_year, 'financial' AS attribute_group,
           attribute_type, attribute_value
    FROM staging.source_financial
),
demographic_attributes AS (
    SELECT customer_id, entry_year, 'demographic' AS attribute_group,
           attribute_type, attribute_value
    FROM staging.source_demographic
)

SELECT customer_id, entry_year, attribute_group, attribute_type, attribute_value
FROM academic_attributes
UNION ALL
SELECT customer_id, entry_year, attribute_group, attribute_type, attribute_value
FROM financial_attributes
UNION ALL
SELECT customer_id, entry_year, attribute_group, attribute_type, attribute_value
FROM demographic_attributes;

Scalable and maintainable: new sources extend, they don't disrupt.

Data-Quality Validation Bundle Pre-release checks that each return rows only on failure, turning "I think it's right" into evidence. 
-- Goal: catch problems before release. Each check returns rows ONLY on failure.

-- 1) Grain uniqueness: no duplicate customer/year
SELECT customer_id, entry_year, COUNT(*) AS record_count
FROM analytics.customer_year_summary
GROUP BY customer_id, entry_year
HAVING COUNT(*) > 1;

-- 2) Not-null on required keys
SELECT COUNT(*) AS null_key_count
FROM analytics.customer_year_summary
WHERE customer_id IS NULL
   OR entry_year IS NULL;

-- 3) Row-count reconciliation vs source of truth
WITH source_count AS (
    SELECT COUNT(DISTINCT customer_id) AS n FROM staging.source_of_truth
),
target_count AS (
    SELECT COUNT(DISTINCT customer_id) AS n FROM analytics.customer_year_summary
)
SELECT s.n AS source_rows, t.n AS target_rows, (t.n - s.n) AS difference
FROM source_count AS s
CROSS JOIN target_count AS t
WHERE ABS(t.n - s.n) > 0;

Evidence-based release confidence across the pipeline.

Identity Resolution (Person Matching) Link records that represent the same person across systems using normalized keys and ranked match rules, producing an ID crosswalk for downstream grouping. 
-- Goal: find record pairs that are the same person across sources,
--       then emit an ID crosswalk that downstream grouping can consume.
-- Pattern: deterministic matching on normalized keys + a match-tier rank.

WITH normalized AS (
    SELECT
        source_system,
        record_id,
        LOWER(TRIM(email_clean))            AS email_key,
        UPPER(TRIM(last_name))              AS last_name_key,
        LEFT(UPPER(TRIM(first_name)), 1)    AS first_initial,
        birth_date
    FROM staging.normalized_persons
),
candidate_pairs AS (
    SELECT
        a.record_id AS id1,
        b.record_id AS id2,
        -- match tier: 1 = strongest signal
        CASE
            WHEN a.email_key = b.email_key                      THEN 1  -- exact email
            WHEN a.last_name_key = b.last_name_key
             AND a.birth_date    = b.birth_date                 THEN 2  -- name + DOB
            WHEN a.last_name_key = b.last_name_key
             AND a.first_initial = b.first_initial
             AND a.birth_date    = b.birth_date                 THEN 3  -- partial name + DOB
        END AS match_tier
    FROM normalized AS a
    JOIN normalized AS b
      ON a.record_id < b.record_id            -- each unordered pair once
     AND a.source_system <> b.source_system    -- only link across systems
)
SELECT id1, id2, match_tier
FROM candidate_pairs
WHERE match_tier IS NOT NULL
-- keep only the strongest tier for each pair
QUALIFY ROW_NUMBER() OVER (
    PARTITION BY id1, id2 ORDER BY match_tier
) = 1;

Deterministic, explainable matches feed the crosswalk that groups a person's records together.

Recursive Graph Walk with Cycle Prevention Group linked identifiers into a single entity by traversing a relationship graph, bounded against cycles and runaway recursion. 
-- Goal: group IDs that are linked (A-B, B-C => {A,B,C}) into one entity.
-- Pattern: recursive CTE with a visited-path array and a depth limit.

WITH RECURSIVE bidirectional_edges AS (
    SELECT id1, id2 FROM staging.id_crosswalk
    UNION
    SELECT id2 AS id1, id1 AS id2 FROM staging.id_crosswalk
),
graph_walk AS (
    -- anchor: start each id with itself
    SELECT
        e.id1 AS root_id,
        e.id1 AS current_id,
        ARRAY_CONSTRUCT(e.id1) AS visited_path,
        0 AS depth
    FROM bidirectional_edges AS e

    UNION ALL

    -- recurse: follow edges, skip visited ids, cap depth
    SELECT
        gw.root_id,
        e.id2 AS current_id,
        ARRAY_APPEND(gw.visited_path, e.id2) AS visited_path,
        gw.depth + 1 AS depth
    FROM graph_walk AS gw
    JOIN bidirectional_edges AS e
      ON gw.current_id = e.id1
    WHERE NOT ARRAY_CONTAINS(e.id2::VARIANT, gw.visited_path)
      AND gw.depth < 50
)
SELECT
    current_id AS member_id,
    MIN(root_id) AS group_id   -- representative id per connected group
FROM graph_walk
GROUP BY current_id;

Finds equivalent records without hard-coding chains.

Email Normalization with REGEXP Standardize and validate free-text email input so downstream matching and deduplication stay reliable. 
-- Goal: clean inconsistent email input and flag invalid addresses.
-- Pattern: REGEXP_REPLACE to strip noise + REGEXP_LIKE to validate format.

SELECT
    raw.contact_id,
    -- lowercase, trim, and remove characters that are never valid in an email
    REGEXP_REPLACE(LOWER(TRIM(raw.email)), '[^a-z0-9._%+\\-@]', '') AS email_clean,
    -- basic structural validation: local@domain.tld
    CASE
        WHEN REGEXP_LIKE(
                 REGEXP_REPLACE(LOWER(TRIM(raw.email)), '[^a-z0-9._%+\\-@]', ''),
                 '^[a-z0-9._%+\\-]+@[a-z0-9.\\-]+\\.[a-z]{2,}$'
             )
        THEN TRUE
        ELSE FALSE
    END AS is_valid_email
FROM staging.raw_contacts AS raw;

Clean, validated keys mean fewer duplicate identities downstream.

Null-Safe Joins & Backfill Join on optional keys without dropping rows, then backfill missing values from a record's neighbors. 
-- Goal: match records on keys that may be NULL, and fill gaps from siblings.
-- Patterns: EQUAL_NULL for null-safe joins; LAG/LEAD IGNORE NULLS to backfill.

WITH joined AS (
    SELECT
        base.customer_id,
        base.entry_year,
        base.funnel_priority,
        COALESCE(base.entry_term, supp.entry_term) AS entry_term
    FROM staging.base_records AS base
    LEFT JOIN staging.supplemental AS supp
        -- EQUAL_NULL: TRUE when both sides match OR both are NULL
        ON EQUAL_NULL(base.customer_id, supp.customer_id)
       AND EQUAL_NULL(base.entry_year, supp.entry_year)
)
SELECT
    customer_id,
    entry_year,
    -- fill missing term from the nearest non-null neighbor for that customer,
    -- then fall back to a sensible default
    COALESCE(
        entry_term,
        LAG(entry_term) IGNORE NULLS OVER (
            PARTITION BY customer_id ORDER BY funnel_priority DESC),
        LEAD(entry_term) IGNORE NULLS OVER (
            PARTITION BY customer_id ORDER BY funnel_priority ASC),
        'Fall'
    ) AS entry_term_filled
FROM joined;

Keeps optional-key data complete and reporting consistent.

Feed Status Rollup with Severity Precedence Collapse many child transaction statuses into one reliable feed-run status using a worst-status-wins rule, with scheduling context. 
-- Goal: one trustworthy status per feed run from many child transactions.
-- Pattern: map statuses to a severity rank, take the worst (MAX), map back.
--          Severity: Timeout (4) > Failed (3) > Reset (2) > Success (1) > No Data (0)

WITH transaction_status AS (
    SELECT
        feed_key,
        feed_run_id,
        DATE(run_scheduled_dt) AS run_date,
        DECODE(INITCAP(run_status),
            'No Data Delivered', 0,
            'Success',           1,
            'Reset',             2,
            'Failed',            3,
            'Timeout',           4
        ) AS severity
    FROM staging.feed_transaction_runs
)
SELECT
    feed_key,
    feed_run_id,
    run_date,
    -- worst child status becomes the feed-run status
    DECODE(MAX(severity),
        0, 'No Data Delivered',
        1, 'Success',
        2, 'Reset',
        3, 'Failed',
        4, 'Timeout'
    ) AS feed_run_status
FROM transaction_status
GROUP BY feed_key, feed_run_id, run_date;

Turns noisy per-transaction logs into clear, monitorable feed health.
